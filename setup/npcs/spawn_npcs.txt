# ZNPCsPlus 2.0.0 NPC Creation Script
# Run from console or RCON (op required).
# Lines starting with # are comments - skip them when pasting.
#
# All NPCs spawn at default coordinates around spawn (0, 101, 0) in world "world".
# Move any NPC after creation:  /npc move <id>           (uses your current pos)
# Or:                            /npc move <id> <x> <y> <z>
# Delete:                        /npc delete <id>
# Inspect:                       /npc list  |  /npc near 50
#
# Skins use Mojang names via:    /npc skin set <id> <name>
# Holograms above the NPC via:   /npc holo add <id> "<text>"
# Click actions via:             /npc action add <id> RUN_COMMAND "<cmd>"
#
# NPCs persist in: plugins/ZNPCsPlus/storage/npcs.yml

# === 1. Spawn Guide NPC ===
npc create spawn_guide PLAYER world 0 101 5
npc skin set spawn_guide Hypixel
npc holo add spawn_guide "&6&lSpawn Guide"
npc holo add spawn_guide "&fRight-click for tour"
npc action add spawn_guide RUN_COMMAND "warp spawn"
